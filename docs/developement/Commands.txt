s//Single Line Commit Precedure:

git add .
git commit -m "comment"
# Update `pyproject.toml` version or use `bump2version/poetry` then:
git tag -a vX.Y.Z -m "release vX.Y.Z"
git push --follow-tags
python -m pip install --user --upgrade build
python -m build

//Detailed Commit Procedure:

git add -p
# Commit with prepared release notes (use -F to supply the prepared changelog file)
git commit -F docs/architecture/00-commit.md
git tag -a vX.Y.Z -m "release v0.0.1"
git push --follow-tags
# Build source+wheel artifacts
python -m pip install --user --upgrade build
python -m build
